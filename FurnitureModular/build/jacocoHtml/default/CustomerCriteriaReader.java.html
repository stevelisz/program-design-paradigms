import java.io.FileNotFoundException;

  private CustomerCriteriaReader customerCriteriaReader;

  public JSONObject getCustomerCriteria() {
    //JSON parser object to parse read file
    JSONParser jsonParser = new JSONParser();
    JSONObject employeeList = new JSONObject();
    JSONObject result = new JSONObject();
    try (FileReader reader = new FileReader("criteria.json"))
    {
      Object obj = jsonParser.parse(reader);

      employeeList = (JSONObject) obj;
      System.out.println(employeeList);


    } catch (FileNotFoundException e) {
      e.printStackTrace();
    } catch (IOException e) {
      e.printStackTrace();
    } catch (ParseException e) {
      e.printStackTrace();
    }

    result = (JSONObject) employeeList.get("criteria");
    return result;
  }
